# `/contacts` (Slice D1) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the read-only `/contacts` Mercury list surface in two PRs (D1a backend, D1b frontend), staying behind `NEXT_PUBLIC_CONTACTS_LIVE=false` until staging review.

**Architecture:** New `packages/core/src/contacts/list.ts` exports `listContactsForBrowse(input, deps)`, a surface-agnostic projection over a new `ContactStore.listForBrowse` method. Fastify exposes `GET /api/dashboard/contacts` via `apps/api/src/routes/dashboard-contacts.ts`; the Next.js dashboard proxies via `apps/dashboard/src/app/api/dashboard/contacts/route.ts`. The page lives at `apps/dashboard/src/app/(auth)/contacts/`, mirrors `/reports`' Mercury vocabulary via the existing `--mercury-*` tokens, and gates live data behind `NEXT_PUBLIC_CONTACTS_LIVE`. Cursor pagination via `(lastActivityAt, id)` keyset; lifecycle-only filter chips; per-contact opportunity count joined in the projection; row click is inert until `ROUTE_AVAILABILITY.contact` flips in D1.5.

**Tech stack:** TypeScript ESM, pnpm + Turborepo, Vitest, Prisma, Fastify, Next.js 14 App Router, NextAuth, TanStack React Query, Zod.

**Spec:** `docs/superpowers/specs/2026-05-08-contacts-d1-design.md` (decisions ledger §2.0 is binding).

## Implementer watch-outs

Three small things to keep in mind during execution:

- **Don't silently widen `ContactStore.list`'s contract.** The plan's `listForBrowse` is a **new** method; adding cursor + search + opportunity-count semantics to the existing `list(orgId, filters)` would change a contract the skill-runtime tools depend on (`packages/core/src/skill-runtime/tools/crm-query.ts`).
- **Cursor encoding is server-only.** The base64 cursor token is opaque to the dashboard. The dashboard does not parse it, must not synthesise it, and must not store it in URL params (the spec explicitly keeps cursor out of the URL — see §6.3).
- **No edits to `EditorialAuthShell`.** D1 ships its own `ContactsHeader` per spec OPEN-4 (locked). Adding a Tools/Contacts link to the editorial header is a future slice.

```ts
async listForBrowse(q: ContactBrowseQuery): Promise<ContactBrowseResult> {
  const { orgId, stage, search, sort, direction, cursor, limit } = q;

  const where: Prisma.ContactWhereInput = { organizationId: orgId };
  if (stage) where.stage = stage;
  if (search) {
    where.OR = [
      { name: { contains: search, mode: "insensitive" } },
      { phone: { contains: search, mode: "insensitive" } },
      { email: { contains: search, mode: "insensitive" } },
    ];
  }
  if (cursor) {
    // Keyset: (ts, id) < cursor for desc; (ts, id) > cursor for asc.
    const cmp = direction === "desc" ? "lt" : "gt";
    const tsField = sort === "lastActivityAt" ? "lastActivityAt" : "firstContactAt";
    where.OR = [
      ...((where.OR as Prisma.ContactWhereInput[] | undefined) ?? []),
      // (tsField, id) < (cursor.ts, cursor.id)  →  tsField < cursor.ts OR (tsField === cursor.ts AND id < cursor.id)
      { [tsField]: { [cmp]: cursor.ts } as Prisma.DateTimeFilter },
      { AND: [{ [tsField]: cursor.ts }, { id: { [cmp]: cursor.id } as Prisma.StringFilter }] },
    ];
    // ⚠ The OR layout above is illustrative; if the existing search OR is
    // present, combine via AND of two ORs (one for search, one for cursor)
    // rather than concatenating. See test "applies cursor + search together".
  }

  const orderBy: Prisma.ContactOrderByWithRelationInput[] = [
    { [sort === "lastActivityAt" ? "lastActivityAt" : "firstContactAt"]: direction },
    { id: direction },
  ];

  const rows = await this.prisma.contact.findMany({
    where,
    orderBy,
    take: limit + 1, // detect hasMore
  });

  const hasMore = rows.length > limit;
  const trimmed = hasMore ? rows.slice(0, limit) : rows;

  // Opportunity counts (non-terminal) per contact. Cap at 99 in code.
  const counts = trimmed.length === 0
    ? new Map<string, number>()
    : await this.fetchOpportunityCounts(trimmed.map((r) => r.id));

  const lastRow = trimmed.at(-1);
  const sortField = sort === "lastActivityAt" ? "lastActivityAt" : "firstContactAt";
  const nextKeyset = hasMore && lastRow
    ? { ts: lastRow[sortField], id: lastRow.id }
    : null;

  return {
    rows: trimmed.map((r) => mapPrismaToContact(r)),
    opportunityCounts: counts,
    hasMore,
    nextKeyset,
  };
}

private async fetchOpportunityCounts(contactIds: string[]): Promise<Map<string, number>> {
  const grouped = await this.prisma.opportunity.groupBy({
    by: ["contactId"],
    where: { contactId: { in: contactIds }, stage: { notIn: ["won", "lost"] } },
    _count: { _all: true },
  });
  const map = new Map<string, number>();
  for (const g of grouped) map.set(g.contactId, Math.min(g._count._all, 99));
  return map;
}
```

Note the search-AND-cursor combinator caveat — the test in Step 1 ("applies cursor + search together") forces this to be correct. Don't ship the illustrative concatenation literally.

Create `packages/core/src/contacts/list.ts`:

```ts
import type { ContactStore, ContactBrowseQuery } from "../lifecycle/contact-store.js";
import type {
  ContactsListQuery,
  ContactsListResponse,
  ContactBrowseRow,
} from "@switchboard/schemas";

export class InvalidCursorError extends Error {
  readonly code = "INVALID_CURSOR";
}

export interface ListContactsDeps {
  contactStore: ContactStore;
}

export async function listContactsForBrowse(
  input: { orgId: string; query: ContactsListQuery },
  deps: ListContactsDeps,
): Promise<ContactsListResponse> {
  const { orgId, query } = input;
  const decoded = query.cursor ? decodeCursor(query.cursor) : undefined;

  const result = await deps.contactStore.listForBrowse({
    orgId,
    stage: query.stage,
    search: query.search,
    sort: query.sort,
    direction: query.direction,
    cursor: decoded,
    limit: query.limit,
  } as ContactBrowseQuery);

  const rows: ContactBrowseRow[] = result.rows.map((c) => ({
    id: c.id,
    displayName: c.name ?? c.phone ?? c.email ?? "—",
    stage: c.stage,
    primaryChannel: c.primaryChannel,
    source: c.source ?? null,
    lastActivityAt: c.lastActivityAt.toISOString(),
    firstContactAt: c.firstContactAt.toISOString(),
    opportunityCount: Math.min(result.opportunityCounts.get(c.id) ?? 0, 99),
    detailHref: `/contacts/${c.id}`,
  }));

  const nextCursor = result.nextKeyset ? encodeCursor(result.nextKeyset) : null;

  return { rows, nextCursor, hasMore: result.hasMore };
}

function encodeCursor(k: { ts: Date; id: string }): string {
  return Buffer.from(JSON.stringify({ ts: k.ts.toISOString(), id: k.id })).toString("base64");
}
function decodeCursor(s: string): { ts: Date; id: string } {
  try {
    const parsed = JSON.parse(Buffer.from(s, "base64").toString("utf8"));
    if (typeof parsed?.ts !== "string" || typeof parsed?.id !== "string") {
      throw new Error("malformed");
    }
    return { ts: new Date(parsed.ts), id: parsed.id };
  } catch {
    throw new InvalidCursorError("Invalid cursor");
  }
}
```

import type { FastifyPluginAsync } from "fastify";
import { ContactsListQuerySchema } from "@switchboard/schemas";
import { listContactsForBrowse, InvalidCursorError } from "@switchboard/core/contacts";
import { requireOrganizationScope } from "../utils/require-org.js";

export const dashboardContactsRoute: FastifyPluginAsync = async (app) => {
  app.get("/api/dashboard/contacts", async (req, reply) => {
    const orgId = await requireOrganizationScope(req, reply);
    if (!orgId) return; // requireOrganizationScope already replied

    const parsed = ContactsListQuerySchema.safeParse(req.query);
    if (!parsed.success) {
      return reply.code(400).send({ error: parsed.error.flatten() });
    }

    try {
      const response = await listContactsForBrowse(
        { orgId, query: parsed.data },
        { contactStore: app.contactStore },
      );
      return response;
    } catch (e) {
      if (e instanceof InvalidCursorError) {
        return reply.code(400).send({ error: "INVALID_CURSOR" });
      }
      throw e;
    }
  });
};
